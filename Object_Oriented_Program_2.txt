            RETURN selected_answer

CLASS ATTEMPT
    ATTRIBUTES
        PRIVATE attempt_id: NUMBER
        PRIVATE student: STUDENT
        PRIVATE exam: EXAM
        PRIVATE start_time: NUMBER
        PRIVATE current_time: NUMBER
        PRIVATE attempt_date: TEXT
        PRIVATE answers: LIST<ANSWER_RECORD>[]

    CONSTRUCTOR(attemptId: NUMBER, attemptStudent: STUDENT, attemptExam: EXAM, startTime: NUMBER, attemptDate: TEXT)
        SET attempt_id TO attemptId
        SET student TO attemptStudent
        SET exam TO attemptExam
        SET start_time TO startTime
        SET current_time TO startTime
        SET attempt_date TO attemptDate
        SET answers TO EMPTY LIST

        CALL student.ADD_ATTEMPT(self)
    END CONSTRUCTOR

    METHODS
        UPDATE_CURRENT_TIME(newTime: NUMBER)
            SET current_time TO newTime
        END METHOD

        IS_TIME_EXPIRED()
            IF current_time - start_time GREATER THAN exam.GET_TIME_LIMIT() THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        ADD_ANSWER(question: QUESTION, selectedAnswer: TEXT)
            IF IS_TIME_EXPIRED() EQUALS TRUE THEN
                DISPLAY "TIME LIMIT PASSED. ANSWER REJECTED"
                RETURN FALSE
            END IF

            IF exam.HAS_QUESTION(question) EQUALS FALSE THEN
                DISPLAY "QUESTION DOES NOT BELONG TO THIS EXAM"
                RETURN FALSE
            END IF

            CREATE OBJECT answerRecord OF ANSWER_RECORD(question, selectedAnswer)
            ADD answerRecord TO answers

            DISPLAY "ANSWER SAVED"
            RETURN TRUE
        END METHOD

        FIND_SELECTED_ANSWER(question: QUESTION)
            REPEAT FROM I = 0 TO SIZE OF answers - 1
                IF answers[I].GET_QUESTION() EQUALS question THEN
                    RETURN answers[I].GET_SELECTED_ANSWER()
                END IF
            END REPEAT

            RETURN ""
        END METHOD

        CALCULATE_SCORE()
            SET score TO 0
            SET examQuestions TO exam.GET_QUESTIONS()

            REPEAT FROM I = 0 TO SIZE OF examQuestions - 1
                SET selectedAnswer TO FIND_SELECTED_ANSWER(examQuestions[I])

                IF selectedAnswer NOT EQUAL TO "" THEN
                    IF examQuestions[I].CHECK_ANSWER(selectedAnswer) EQUALS TRUE THEN
                        SET score TO score + examQuestions[I].GET_POINT_VALUE()
                    END IF
                END IF
            END REPEAT

            RETURN score
        END METHOD

        DESCRIBE()
            DISPLAY "Exam: " + exam.GET_TITLE()
            DISPLAY "Attempt Date: " + attempt_date
            DISPLAY "Score: " + CALCULATE_SCORE()
        END METHOD
END CLASS


START
    BEGIN
        CREATE OBJECT exam1 OF EXAM(1, "OOP Exam", 60)

        CREATE OBJECT question1 OF QUESTION(101, "What is encapsulation?", "Data hiding", 10)
        CREATE OBJECT question2 OF QUESTION(102, "What is a class?", "Blueprint", 10)

        CALL exam1.ADD_QUESTION(question1)
        CALL exam1.ADD_QUESTION(question2)

        CREATE OBJECT student1 OF STUDENT(201, "Ali")

        CREATE OBJECT attempt1 OF ATTEMPT(301, student1, exam1, 0, "2026-07-07")

        CALL attempt1.ADD_ANSWER(question1, "Data hiding")
        CALL attempt1.ADD_ANSWER(question2, "Wrong answer")

        CALL student1.VIEW_ATTEMPTS()

        DISPLAY attempt1.CALCULATE_SCORE()
    END
STOP

CLASS TRIP
    ATTRIBUTES
        PRIVATE trip_id: NUMBER
        PRIVATE passenger: PASSENGER
        PRIVATE driver: DRIVER
        PRIVATE pickup_location: TEXT
        PRIVATE dropoff_location: TEXT
        PRIVATE fare: NUMBER
        PRIVATE status: TEXT

    CONSTRUCTOR(tripId: NUMBER, tripPassenger: PASSENGER, tripDriver: DRIVER, pickup: TEXT, dropoff: TEXT, tripFare: NUMBER)
        SET trip_id TO tripId
        SET passenger TO tripPassenger
        SET driver TO tripDriver

        IF tripFare GREATER THAN 0 AND pickup NOT EQUAL TO dropoff THEN
            SET pickup_location TO pickup
            SET dropoff_location TO dropoff
            SET fare TO tripFare
            SET status TO "CREATED"

            CALL passenger.ADD_TRIP(self)

        ELSE
            SET pickup_location TO pickup
            SET dropoff_location TO dropoff
            SET fare TO 0
            SET status TO "INVALID"
            DISPLAY "INVALID TRIP DETAILS"
        END IF
    END CONSTRUCTOR

    METHODS
        COMPLETE_TRIP()
            IF status EQUALS "CREATED" THEN
                SET status TO "COMPLETED"
                DISPLAY "TRIP COMPLETED"
            ELSE
                DISPLAY "TRIP CANNOT BE COMPLETED"
            END IF
        END METHOD

        GET_FARE()
            RETURN fare
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        DESCRIBE()
            DISPLAY "Driver: " + driver.GET_DRIVER_NAME()
            DISPLAY "Passenger: " + passenger.GET_PASSENGER_NAME()
            DISPLAY "Pickup: " + pickup_location
            DISPLAY "Drop-off: " + dropoff_location
            DISPLAY "Fare: " + fare
            DISPLAY "Status: " + status
        END METHOD

CLASS MEMBER
    ATTRIBUTES
        PRIVATE member_id: NUMBER
        PRIVATE name: TEXT
        PRIVATE phone: TEXT
        PRIVATE membership_plan: MEMBERSHIP_PLAN
        PRIVATE bookings: LIST<BOOKING>[]

    CONSTRUCTOR(memberId: NUMBER, memberName: TEXT, memberPhone: TEXT, plan: MEMBERSHIP_PLAN)
        SET member_id TO memberId
        SET name TO memberName
        SET phone TO memberPhone
        SET membership_plan TO plan
        SET bookings TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        GET_MEMBER_NAME()
            RETURN name
        END METHOD

        ADD_BOOKING(newBooking: BOOKING)
            ADD newBooking TO bookings
            DISPLAY "BOOKING ADDED TO MEMBER"
        END METHOD

        CALCULATE_USED_HOURS(monthName: TEXT)
            SET usedHours TO 0

            REPEAT FROM I = 0 TO SIZE OF bookings - 1
                IF bookings[I].GET_MONTH() EQUALS monthName THEN
                    SET usedHours TO usedHours + bookings[I].CALCULATE_DURATION()
                END IF
            END REPEAT

            RETURN usedHours
        END METHOD

        CALCULATE_REMAINING_HOURS(monthName: TEXT)
            RETURN membership_plan.GET_MONTHLY_QUOTA() - CALCULATE_USED_HOURS(monthName)
        END METHOD

        CAN_BOOK(duration: NUMBER, monthName: TEXT)
            IF CALCULATE_USED_HOURS(monthName) + duration GREATER THAN membership_plan.GET_MONTHLY_QUOTA() THEN
                RETURN FALSE
            ELSE
                RETURN TRUE
            END IF
        END METHOD
END CLASS

CLASS BOOKING
    ATTRIBUTES
        PRIVATE booking_id: NUMBER
        PRIVATE member: MEMBER
        PRIVATE room: ROOM
        PRIVATE booking_date: TEXT
        PRIVATE month: TEXT
        PRIVATE start_time: NUMBER
        PRIVATE end_time: NUMBER
        PRIVATE status: TEXT

    CONSTRUCTOR(bookingId: NUMBER, bookingMember: MEMBER, bookingRoom: ROOM, dateValue: TEXT, monthName: TEXT, startTime: NUMBER, endTime: NUMBER)
        SET booking_id TO bookingId
        SET member TO bookingMember
        SET room TO bookingRoom
        SET booking_date TO dateValue
        SET month TO monthName

        IF startTime LESS THAN endTime THEN
            SET start_time TO startTime
            SET end_time TO endTime
            SET status TO "PENDING"
        ELSE
            SET start_time TO 0
            SET end_time TO 0
            SET status TO "INVALID"
            DISPLAY "INVALID BOOKING TIME"
        END IF
    END CONSTRUCTOR

    METHODS
        CALCULATE_DURATION()
            RETURN end_time - start_time
        END METHOD

        GET_MONTH()
            RETURN month
        END METHOD

        GET_DATE()
            RETURN booking_date
        END METHOD

        GET_START_TIME()
            RETURN start_time
        END METHOD

        GET_END_TIME()
            RETURN end_time
        END METHOD

        CHECK_OVERLAP(otherBooking: BOOKING)
            IF booking_date EQUALS otherBooking.GET_DATE() THEN
                IF start_time LESS THAN otherBooking.GET_END_TIME() AND end_time GREATER THAN otherBooking.GET_START_TIME() THEN
                    RETURN TRUE
                END IF
            END IF

            RETURN FALSE
        END METHOD

        CONFIRM()
            IF status EQUALS "INVALID" THEN
                DISPLAY "BOOKING REJECTED: INVALID BOOKING"
                RETURN FALSE
            END IF

            IF member.CAN_BOOK(CALCULATE_DURATION(), month) EQUALS FALSE THEN
                DISPLAY "BOOKING REJECTED: QUOTA EXCEEDED"
                SET status TO "REJECTED"
                RETURN FALSE
            END IF

            IF room.IS_AVAILABLE(self) EQUALS FALSE THEN
                DISPLAY "BOOKING REJECTED: ROOM NOT AVAILABLE"
                SET status TO "REJECTED"
                RETURN FALSE
            END IF

            CALL member.ADD_BOOKING(self)
            CALL room.ADD_BOOKING(self)

            SET status TO "CONFIRMED"
            DISPLAY "BOOKING CONFIRMED"
            RETURN TRUE
        END METHOD

        DESCRIBE()
            DISPLAY "Member: " + member.GET_MEMBER_NAME()
            DISPLAY "Room: " + room.GET_ROOM_NAME()
            DISPLAY "Date: " + booking_date
            DISPLAY "Start: " + start_time
            DISPLAY "End: " + end_time
            DISPLAY "Status: " + status
        END METHOD
END CLASS
