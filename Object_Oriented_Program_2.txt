CLASS Attempt
    ATTRIBUTES
        PRIVATE attemptID : Number
        PRIVATE exam : Exam
        PRIVATE student : Student
        PRIVATE selectedAnswers : Map<Question, String>
        PRIVATE startTime : Number
    CONSTRUCTOR(attemptID, exam, student, startTime)
        this.attemptID = attemptID
        this.exam = exam
        this.student = student
        this.startTime = startTime
        this.selectedAnswers = NEW Map<Question, String>()
    END CONSTRUCTOR
    METHOD submitAnswer(question, answer, currentTime)
        IF currentTime - this.startTime <= this.exam.timeLimit THEN
            ADD answer TO this.selectedAnswers WITH question
        ELSE
            DISPLAY "Time limit exceeded"
        END IF
    END METHOD
    METHOD calculateScore()
        SET score = 0

        FOR EACH question IN this.exam.getQuestions()
            IF question.checkAnswer(this.selectedAnswers[question]) THEN
                SET score = score + question.getPoints()
            END IF
        END FOR
        RETURN score
    END METHOD
    METHOD getExam()
        RETURN this.exam
    END METHOD
END CLASS

  END

CLASS Room
    ATTRIBUTES
        PRIVATE roomID : Number
        PRIVATE roomName : String
        PRIVATE bookings : List<Booking>
    CONSTRUCTOR(roomID, roomName)
        this.roomID = roomID
        this.roomName = roomName
        this.bookings = NEW List<Booking>()
    END CONSTRUCTOR
    METHOD isAvailable(newStartTime, newEndTime)
        FOR EACH booking IN this.bookings
            IF booking.overlaps(newStartTime, newEndTime) THEN
                RETURN FALSE
            END IF
        END FOR
        RETURN TRUE
    END METHOD

    ATTRIBUTES
        PRIVATE bookingID : Number
        PRIVATE member : Member
        PRIVATE room : Room
        PRIVATE bookingDate : String
        PRIVATE startTime : Number
        PRIVATE endTime : Number
        PRIVATE duration : Number
    CONSTRUCTOR(bookingID, member, room, bookingDate, startTime, endTime, duration)
        IF startTime < endTime AND duration > 0 THEN
            this.bookingID = bookingID
            this.member = member
            this.room = room
            this.bookingDate = bookingDate
            this.startTime = startTime
            this.endTime = endTime
            this.duration = duration
        END IF
    END CONSTRUCTOR
    METHOD overlaps(newStartTime, newEndTime)
        RETURN newStartTime < this.endTime AND newEndTime > this.startTime
    END METHOD
    METHOD getDuration()
        RETURN this.duration
    END METHOD
    METHOD confirmBooking()
        IF this.member.getRemainingHours() >= this.duration AND this.room.isAvailable(this.startTime, this.endTime) THEN
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD
