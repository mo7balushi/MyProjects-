CLASS SPECIES
    ATTRIBUTES
        PRIVATE species_name: TEXT
        PRIVATE care_requirements: TEXT
        PRIVATE safe_temperature_min: NUMBER
        PRIVATE safe_temperature_max: NUMBER
        PRIVATE safe_heart_rate_min: NUMBER
        PRIVATE safe_heart_rate_max: NUMBER

    CONSTRUCTOR(speciesName: TEXT, careReq: TEXT, tempMin: NUMBER, tempMax: NUMBER, heartMin: NUMBER, heartMax: NUMBER)
        SET species_name TO speciesName
        SET care_requirements TO careReq
        SET safe_temperature_min TO tempMin
        SET safe_temperature_max TO tempMax
        SET safe_heart_rate_min TO heartMin
        SET safe_heart_rate_max TO heartMax
    END CONSTRUCTOR

    METHODS

        GET_SPECIES_NAME()
            DISPLAY species_name
        END METHOD

        GET_CARE_REQUIREMENTS()
            DISPLAY care_requirements
        END METHOD

        CHECK_TEMPERATURE(temp: NUMBER)
            IF temp LESS THAN safe_temperature_min OR temp GREATER THAN safe_temperature_max THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        CHECK_HEART_RATE(rate: NUMBER)
            IF rate LESS THAN safe_heart_rate_min OR rate GREATER THAN safe_heart_rate_max THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
END CLASS

        SET temperature TO temp
        SET heart_rate TO heartRate
        SET treatments TO EMPTY LIST
        SET urgent TO FALSE
    END CONSTRUCTOR

    METHODS

        ADD_TREATMENT(newTreatment: TREATMENT)
            ADD newTreatment TO treatments
            DISPLAY "TREATMENT ADDED"
        END METHOD

        CHECK_URGENCY(petSpecies: SPECIES)
            IF petSpecies.CHECK_TEMPERATURE(temperature) EQUALS TRUE OR petSpecies.CHECK_HEART_RATE(heart_rate) EQUALS TRUE THEN
                SET urgent TO TRUE
                DISPLAY "URGENT EXAMINATION"
            ELSE
                SET urgent TO FALSE
                DISPLAY "NORMAL EXAMINATION"
            END IF
        END METHOD

        GET_TREATMENTS()
            DISPLAY treatments
        END METHOD

        GET_URGENCY()
            DISPLAY urgent
        END METHOD

        CALCULATE_TREATMENT_TOTAL()
            SET total TO 0

            REPEAT FROM I = 0 TO SIZE OF treatments - 1
                SET total TO total + treatments[I].GET_COST()
            END REPEAT

            RETURN total
        END METHOD

ATTRIBUTES

PRIVATE end_time: NUMBER

    SET film TO showFilm
    SET screen TO showScreen
    SET start_time TO startTime
    SET end_time TO endTime
    SET booked_seats TO EMPTY LIST
END CONSTRUCTOR
METHODS
    CHECK_OVERLAP(otherShowtime: SHOWTIME)
        IF start_time LESS THAN otherShowtime.end_time AND end_time GREATER THAN otherShowtime.start_time THEN
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD
    BOOK_SEAT(seatNumber: TEXT)
        IF seatNumber NOT IN screen.seats THEN
            DISPLAY "SEAT DOES NOT EXIST"
            RETURN FALSE
        END IF
        IF seatNumber IN booked_seats THEN
            DISPLAY "SEAT ALREADY BOOKED"
            RETURN FALSE
        ELSE
            ADD seatNumber TO booked_seats
            DISPLAY "SEAT BOOKED SUCCESSFULLY"
            RETURN TRUE
        END IF
    END METHOD
    GET_AVAILABLE_SEATS()
        SET available TO screen.capacity - SIZE OF booked_seats
        DISPLAY available
    END METHOD
    GET_FILM()
        DISPLAY film
    END METHOD
    GET_SCREEN()
        DISPLAY screen
    END METHOD
END CLASS

CLASS VENDOR
ATTRIBUTES
PRIVATE vendor_id: NUMBER
PRIVATE name: TEXT
PRIVATE phone: TEXT
PRIVATE email: TEXT
PRIVATE late_deliveries: NUMBER
PRIVATE total_deliveries: NUMBER
PRIVATE reliability_rating: NUMBER
PRIVATE products: LIST<PRODUCT>[]


CONSTRUCTOR(vendorId: NUMBER, vendorName: TEXT, vendorPhone: TEXT, vendorEmail: TEXT)
    SET vendor_id TO vendorId
    SET name TO vendorName
    SET phone TO vendorPhone
    SET email TO vendorEmail
    SET late_deliveries TO 0
    SET total_deliveries TO 0
    SET reliability_rating TO 100
    SET products TO EMPTY LIST
END CONSTRUCTOR

METHODS

    ADD_PRODUCT(newProduct: PRODUCT)
        ADD newProduct TO products
        DISPLAY "PRODUCT ADDED TO VENDOR"
    END METHOD

    RECORD_DELIVERY(isLate: BOOLEAN)
        SET total_deliveries TO total_deliveries + 1

        IF isLate EQUALS TRUE THEN
            SET late_deliveries TO late_deliveries + 1
        END IF

        CALL UPDATE_RELIABILITY()
    END METHOD

    UPDATE_RELIABILITY()
        IF total_deliveries EQUALS 0 THEN
            SET reliability_rating TO 100
        ELSE
            SET reliability_rating TO 100 - ((late_deliveries / total_deliveries) * 100)
        END IF
    END METHOD

    GET_RELIABILITY()
        DISPLAY reliability_rating
    END METHOD

    GET_PRODUCTS()
        DISPLAY products
    END METHOD


END CLASS

CLASS STORAGE_BIN
ATTRIBUTES
PRIVATE bin_id: NUMBER
PRIVATE location: TEXT
PRIVATE quantity: NUMBER


CONSTRUCTOR(binId: NUMBER, binLocation: TEXT, startQuantity: NUMBER)
    SET bin_id TO binId
    SET location TO binLocation

    IF startQuantity LESS THAN 0 THEN
        SET quantity TO 0
    ELSE
        SET quantity TO startQuantity
    END IF
END CONSTRUCTOR

METHODS

    GET_QUANTITY()
        RETURN quantity
    END METHOD

    ADD_STOCK(amount: NUMBER)
        IF amount GREATER THAN 0 THEN
            SET quantity TO quantity + amount
            DISPLAY "STOCK ADDED TO BIN"
        ELSE
            DISPLAY "INVALID STOCK AMOUNT"
        END IF
    END METHOD

    REMOVE_STOCK(amount: NUMBER)
        IF amount LESS THAN OR EQUAL TO 0 THEN
            DISPLAY "INVALID STOCK AMOUNT"
            RETURN FALSE
        ELSE
            IF amount GREATER THAN quantity THEN
                DISPLAY "NOT ENOUGH STOCK IN BIN"
                RETURN FALSE
            ELSE
                SET quantity TO quantity - amount
                DISPLAY "STOCK REMOVED FROM BIN"
                RETURN TRUE
            END IF
        END IF
    END METHOD


END CLASS

CLASS PRODUCT
ATTRIBUTES
PRIVATE product_id: NUMBER
PRIVATE name: TEXT
PRIVATE category: CATEGORY
PRIVATE vendor: VENDOR
PRIVATE storage_bins: LIST<STORAGE_BIN>[]
PRIVATE stock_movements: LIST<STOCK_MOVEMENT>[]


CONSTRUCTOR(productId: NUMBER, productName: TEXT, productCategory: CATEGORY, productVendor: VENDOR)
    SET product_id TO productId
    SET name TO productName
    SET category TO productCategory
    SET vendor TO productVendor
    SET storage_bins TO EMPTY LIST
    SET stock_movements TO EMPTY LIST
END CONSTRUCTOR

METHODS

    ADD_BIN(newBin: STORAGE_BIN)
        ADD newBin TO storage_bins
        DISPLAY "BIN ADDED TO PRODUCT"
    END METHOD

    CALCULATE_TOTAL_STOCK()
        SET total TO 0

        REPEAT FROM I = 0 TO SIZE OF storage_bins - 1
            SET total TO total + storage_bins[I].GET_QUANTITY()
        END REPEAT

        RETURN total
    END METHOD

    MOVE_STOCK_IN(binIndex: NUMBER, amount: NUMBER, movementId: NUMBER, moveDate: TEXT)
        IF amount LESS THAN OR EQUAL TO 0 THEN
            DISPLAY "INVALID STOCK AMOUNT"
        ELSE
            CALL storage_bins[binIndex].ADD_STOCK(amount)

            CREATE OBJECT movement OF STOCK_MOVEMENT(movementId, self, "IN", amount, moveDate)

            ADD movement TO stock_movements

            DISPLAY "STOCK IN MOVEMENT RECORDED"
        END IF
    END METHOD

    MOVE_STOCK_OUT(binIndex: NUMBER, amount: NUMBER, movementId: NUMBER, moveDate: TEXT)
        IF amount LESS THAN OR EQUAL TO 0 THEN
            DISPLAY "INVALID STOCK AMOUNT"
        ELSE
            IF amount GREATER THAN CALCULATE_TOTAL_STOCK() THEN
                DISPLAY "OUTGOING MOVEMENT REJECTED: STOCK CANNOT GO BELOW ZERO"
            ELSE
                IF storage_bins[binIndex].REMOVE_STOCK(amount) EQUALS TRUE THEN
                    CREATE OBJECT movement OF STOCK_MOVEMENT(movementId, self, "OUT", amount, moveDate)

                    ADD movement TO stock_movements

                    DISPLAY "STOCK OUT MOVEMENT RECORDED"
                END IF
            END IF
        END IF
    END METHOD

    GET_TOTAL_STOCK()
        DISPLAY CALCULATE_TOTAL_STOCK()
    END METHOD

    GET_MOVEMENTS()
        DISPLAY stock_movements
    END METHOD


END CLASS
