CLASS SHOWTIME
    ATTRIBUTES
        PRIVATE showtime_id: NUMBER
        PRIVATE film: FILM
        PRIVATE screen: SCREEN
        PRIVATE start_time: NUMBER
        PRIVATE end_time: NUMBER
        PRIVATE booked_seats: LIST<TEXT>[]
    CONSTRUCTOR(showtimeId: NUMBER, showFilm: FILM, showScreen: SCREEN, startTime: NUMBER, endTime: NUMBER)
        SET showtime_id TO showtimeId
        SET film TO showFilm
        SET screen TO showScreen
        SET start_time TO startTime
        SET end_time TO endTime
        SET booked_seats TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        CHECK_OVERLAP(otherShowtime: SHOWTIME)
            IF start_time LESS THAN otherShowtime.end_time AND end_time GREATER THAN otherShowtime.start_time THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        BOOK_SEAT(seatNumber: TEXT)
            IF seatNumber NOT IN screen.GET_SEATS() THEN
                DISPLAY "SEAT DOES NOT EXIST"
                RETURN FALSE
            END IF
            IF seatNumber IN booked_seats THEN
                DISPLAY "SEAT ALREADY BOOKED"
                RETURN FALSE
            ELSE
                ADD seatNumber TO booked_seats
                DISPLAY "SEAT BOOKED SUCCESSFULLY"
                RETURN TRUE
            END IF
        END METHOD
        GET_AVAILABLE_SEATS()
            RETURN screen.GET_CAPACITY() - SIZE OF booked_seats
        END METHOD
        GET_FILM()
            RETURN film
        END METHOD
        GET_SCREEN()
            RETURN screen
        END METHOD
END CLASS

TASK 280: Warehouse Inventory
CLASS VENDOR
    ATTRIBUTES
        PRIVATE vendor_id: NUMBER
        PRIVATE name: TEXT
        PRIVATE phone: TEXT
        PRIVATE email: TEXT
        PRIVATE late_deliveries: NUMBER
        PRIVATE total_deliveries: NUMBER
        PRIVATE reliability_rating: NUMBER
        PRIVATE products: LIST<PRODUCT>[]
    CONSTRUCTOR(vendorId: NUMBER, vendorName: TEXT, vendorPhone: TEXT, vendorEmail: TEXT)
        SET vendor_id TO vendorId
        SET name TO vendorName
        SET phone TO vendorPhone
        SET email TO vendorEmail
        SET late_deliveries TO 0
        SET total_deliveries TO 0
        SET reliability_rating TO 100
        SET products TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        ADD_PRODUCT(newProduct: PRODUCT)
            ADD newProduct TO products
        END METHOD
        RECORD_DELIVERY(isLate: BOOLEAN)
            SET total_deliveries TO total_deliveries + 1
            IF isLate EQUALS TRUE THEN
                SET late_deliveries TO late_deliveries + 1
            END IF
            CALL UPDATE_RELIABILITY()
        END METHOD
        UPDATE_RELIABILITY()
            IF total_deliveries EQUALS 0 THEN
                SET reliability_rating TO 100
            ELSE
                SET reliability_rating TO 100 - ((late_deliveries / total_deliveries) * 100)
            END IF
        END METHOD
        GET_RELIABILITY()
            RETURN reliability_rating
        END METHOD
END CLASS

    ATTRIBUTES

        SET bin_id TO binId
        SET location TO binLocation

        IF startQuantity LESS THAN 0 THEN
            SET quantity TO 0
        ELSE
            SET quantity TO startQuantity
        END IF
    END CONSTRUCTOR
    METHODS
        GET_QUANTITY()
            RETURN quantity
        END METHOD
        ADD_STOCK(amount: NUMBER)
            IF amount GREATER THAN 0 THEN
                SET quantity TO quantity + amount
                RETURN TRUE
            ELSE
                DISPLAY "INVALID STOCK AMOUNT"
                RETURN FALSE
            END IF
        END METHOD
        REMOVE_STOCK(amount: NUMBER)
            IF amount LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "INVALID STOCK AMOUNT"
                RETURN FALSE
            END IF
            IF amount GREATER THAN quantity THEN
                DISPLAY "NOT ENOUGH STOCK IN BIN"
                RETURN FALSE
            ELSE
                SET quantity TO quantity - amount
                RETURN TRUE
            END IF
        END METHOD
END CLASS

CLASS PRODUCT
    ATTRIBUTES
        PRIVATE product_id: NUMBER
        PRIVATE name: TEXT
        PRIVATE category: CATEGORY
        PRIVATE vendor: VENDOR
        PRIVATE storage_bins: LIST<STORAGE_BIN>[]
        PRIVATE stock_movements: LIST<STOCK_MOVEMENT>[]
    CONSTRUCTOR(productId: NUMBER, productName: TEXT, productCategory: CATEGORY, productVendor: VENDOR)
        SET product_id TO productId
        SET name TO productName
        SET category TO productCategory
        SET vendor TO productVendor
        SET storage_bins TO EMPTY LIST
        SET stock_movements TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        ADD_BIN(newBin: STORAGE_BIN)
            ADD newBin TO storage_bins
        END METHOD
        VALIDATE_BIN_INDEX(binIndex: NUMBER)
            IF binIndex LESS THAN 0 OR binIndex GREATER THAN OR EQUAL TO SIZE OF storage_bins THEN
                DISPLAY "INVALID BIN INDEX"
                RETURN FALSE
            ELSE
                RETURN TRUE
            END IF
        END METHOD
        CALCULATE_TOTAL_STOCK()
            SET total TO 0
            REPEAT FROM I = 0 TO SIZE OF storage_bins - 1
                SET total TO total + storage_bins[I].GET_QUANTITY()
            END REPEAT
            RETURN total
        END METHOD
        MOVE_STOCK_IN(binIndex: NUMBER, amount: NUMBER, movementId: NUMBER, moveDate: TEXT)
            IF VALIDATE_BIN_INDEX(binIndex) EQUALS FALSE THEN
                RETURN FALSE
            END IF
            IF amount LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "INVALID STOCK AMOUNT"
                RETURN FALSE
            END IF
            CALL storage_bins[binIndex].ADD_STOCK(amount)
            CREATE OBJECT movement OF STOCK_MOVEMENT(movementId, self, "IN", amount, moveDate)
            ADD movement TO stock_movements
            DISPLAY "STOCK IN MOVEMENT RECORDED"
            RETURN TRUE
        END METHOD
        MOVE_STOCK_OUT(binIndex: NUMBER, amount: NUMBER, movementId: NUMBER, moveDate: TEXT)
            IF VALIDATE_BIN_INDEX(binIndex) EQUALS FALSE THEN
                RETURN FALSE
            END IF
            IF amount LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "INVALID STOCK AMOUNT"
                RETURN FALSE
            END IF
            IF amount GREATER THAN CALCULATE_TOTAL_STOCK() THEN
                DISPLAY "OUTGOING MOVEMENT REJECTED: STOCK CANNOT GO BELOW ZERO"
                RETURN FALSE
            END IF
            IF storage_bins[binIndex].REMOVE_STOCK(amount) EQUALS TRUE THEN
                CREATE OBJECT movement OF STOCK_MOVEMENT(movementId, self, "OUT", amount, moveDate)
                ADD movement TO stock_movements
                DISPLAY "STOCK OUT MOVEMENT RECORDED"
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        GET_TOTAL_STOCK()
            RETURN CALCULATE_TOTAL_STOCK()
        END METHOD
END CLASS

    BEGIN
        CREATE OBJECT vendor1 OF VENDOR(1, "Oman Supplies", "91261161", "sales@omansupplies.com")
        CREATE OBJECT category1 OF CATEGORY(101, "Electronics", "Electronic warehouse products")
        CREATE OBJECT product1 OF PRODUCT(201, "Keyboard", category1, vendor1)
        CALL vendor1.ADD_PRODUCT(product1)
        CREATE OBJECT bin1 OF STORAGE_BIN(301, "A1", 20)
        CREATE OBJECT bin2 OF STORAGE_BIN(302, "A2", 15)
        CALL product1.ADD_BIN(bin1)
        CALL product1.ADD_BIN(bin2)
        CALL product1.MOVE_STOCK_IN(0, 10, 401, "2026-06-30")
        CALL product1.MOVE_STOCK_OUT(1, 5, 402, "2026-06-30")
        CALL product1.MOVE_STOCK_OUT(0, 100, 403, "2026-06-30")
        CALL vendor1.RECORD_DELIVERY(FALSE)
        CALL vendor1.RECORD_DELIVERY(TRUE)
        DISPLAY product1.GET_TOTAL_STOCK()
        DISPLAY vendor1.GET_RELIABILITY()

        PRIVATE enrolled_classes: LIST<SCHOOL_CLASS>[]
    CONSTRUCTOR(studentId: NUMBER, studentName: TEXT, record: TEXT)
        SET student_id TO studentId
        SET name TO studentName
        SET personal_record TO record
        SET grades TO EMPTY LIST
        SET enrolled_classes TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        ADD_CLASS(newClass: SCHOOL_CLASS)
            ADD newClass TO enrolled_classes
        END METHOD
        HAS_TIME_CONFLICT(newClass: SCHOOL_CLASS)
            REPEAT FROM I = 0 TO SIZE OF enrolled_classes - 1
                IF enrolled_classes[I].GET_SCHEDULE_TIME() EQUALS newClass.GET_SCHEDULE_TIME() THEN
                    RETURN TRUE
                END IF
            END REPEAT
            RETURN FALSE
        END METHOD
        ADD_GRADE(newGrade: GRADE)
            ADD newGrade TO grades
        END METHOD
        CALCULATE_AVERAGE()
            SET total TO 0
            SET totalWeight TO 0
            REPEAT FROM I = 0 TO SIZE OF grades - 1
                SET total TO total + grades[I].GET_WEIGHTED_SCORE()
                SET totalWeight TO totalWeight + grades[I].GET_WEIGHTING()
            END REPEAT

            IF totalWeight EQUALS 0 THEN
                RETURN 0
            ELSE
                RETURN total / totalWeight
            END IF
        END METHOD
END CLASS

                RETURN TRUE

        PRIVATE guests: LIST<GUEST>[]

    CONSTRUCTOR(eventId: NUMBER, eventCouple: COUPLE, eventDate: TEXT, eventBudget: NUMBER)

        ELSE

        SET guests TO EMPTY LIST
    END CONSTRUCTOR

                SET total TO total + tasks[I].GET_COST()
            END REPEAT

            RETURN total
        END METHOD

        ADD_TASK(newTask: PLANNING_TASK)
            IF newTask.GET_VENDOR().IS_AVAILABLE(event_date) EQUALS FALSE THEN
                DISPLAY "VENDOR ALREADY BOOKED"
                RETURN FALSE
            END IF

            IF CALCULATE_TOTAL_SPENDING() + newTask.GET_COST() GREATER THAN budget THEN
                DISPLAY "TASK EXCEEDS BUDGET"
                RETURN FALSE
            END IF

            ADD newTask TO tasks
            CALL newTask.GET_VENDOR().BOOK_DATE(event_date)

            REPEAT FROM I = 0 TO SIZE OF tasks - 1
                IF tasks[I].IS_DONE() EQUALS TRUE THEN
                    SET doneCount TO doneCount + 1
                END IF
            END REPEAT

            RETURN (doneCount / SIZE OF tasks) * 100
        END METHOD

        COUNT_ACTIVITIES_ON_DATE(activityDate: TEXT)
            SET count TO 0

            REPEAT FROM I = 0 TO SIZE OF activity_dates - 1
                IF activity_dates[I] EQUALS activityDate THEN
                    SET count TO count + 1
                END IF
            END REPEAT

            RETURN count
        END METHOD

        CAN_ASSIGN_ACTIVITY(activityDate: TEXT)
            IF COUNT_ACTIVITIES_ON_DATE(activityDate) LESS THAN daily_limit THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        ASSIGN_ACTIVITY(activityDate: TEXT)
            IF CAN_ASSIGN_ACTIVITY(activityDate) EQUALS TRUE THEN
                ADD activityDate TO activity_dates
                RETURN TRUE
            ELSE
                DISPLAY "WORKER DAILY LIMIT REACHED"
                RETURN FALSE
            END IF
        END METHOD
END CLASS


CLASS ACTIVITY
    ATTRIBUTES
        PRIVATE activity_id: NUMBER
        PRIVATE activity_type: TEXT
        PRIVATE amount: NUMBER
        PRIVATE worker: WORKER
        PRIVATE activity_date: TEXT

    CONSTRUCTOR(activityId: NUMBER, typeName: TEXT, activityAmount: NUMBER, activityWorker: WORKER, activityDate: TEXT)
        SET activity_id TO activityId
        SET activity_type TO typeName
        SET amount TO activityAmount
        SET worker TO activityWorker
        SET activity_date TO activityDate
    END CONSTRUCTOR

    METHODS
        GET_TYPE()
            RETURN activity_type
        END METHOD

        GET_AMOUNT()
            RETURN amount
        END METHOD

        GET_WORKER()
            RETURN worker
        END METHOD

        GET_DATE()
            RETURN activity_date
        END METHOD
END CLASS


CLASS FIELD
    ATTRIBUTES
        PRIVATE field_id: NUMBER
        PRIVATE crop: CROP
        PRIVATE activities: LIST<ACTIVITY>[]

    CONSTRUCTOR(fieldId: NUMBER, fieldCrop: CROP)
        SET field_id TO fieldId
        SET crop TO fieldCrop
        SET activities TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_ACTIVITY(activity: ACTIVITY)
            IF activity.GET_WORKER().ASSIGN_ACTIVITY(activity.GET_DATE()) EQUALS FALSE THEN
                DISPLAY "ACTIVITY REJECTED"
                RETURN FALSE
            END IF

            IF activity.GET_TYPE() EQUALS "WATERING" THEN
                IF activity.GET_AMOUNT() LESS THAN crop.GET_WATER_NEED() THEN
                    DISPLAY "WARNING: WATER BELOW CROP NEED"
                END IF
            END IF

            ADD activity TO activities
            DISPLAY "ACTIVITY ADDED"
            RETURN TRUE
        END METHOD

        ESTIMATE_YIELD()
            SET careScore TO 0

            REPEAT FROM I = 0 TO SIZE OF activities - 1
                IF activities[I].GET_TYPE() EQUALS "WATERING" AND activities[I].GET_AMOUNT() GREATER THAN OR EQUAL TO crop.GET_WATER_NEED() THEN
                    SET careScore TO careScore + 1
                END IF
            END REPEAT

            IF careScore GREATER THAN OR EQUAL TO 3 THEN
                RETURN crop.GET_EXPECTED_OUTPUT()
            ELSE IF careScore EQUALS 2 THEN
                RETURN crop.GET_EXPECTED_OUTPUT() * 0.80
            ELSE IF careScore EQUALS 1 THEN
                RETURN crop.GET_EXPECTED_OUTPUT() * 0.60
            ELSE
                RETURN crop.GET_EXPECTED_OUTPUT() * 0.40
            END IF
        END METHOD
END CLASS


CLASS FARM
    ATTRIBUTES
        PRIVATE farm_id: NUMBER
        PRIVATE name: TEXT
        PRIVATE fields: LIST<FIELD>[]
        PRIVATE workers: LIST<WORKER>[]

    CONSTRUCTOR(farmId: NUMBER, farmName: TEXT)
        SET farm_id TO farmId
        SET name TO farmName
        SET fields TO EMPTY LIST
        SET workers TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_FIELD(field: FIELD)
            ADD field TO fields
            DISPLAY "FIELD ADDED"
        END METHOD

        ADD_WORKER(worker: WORKER)
            ADD worker TO workers
            DISPLAY "WORKER ADDED"
        END METHOD
END CLASS
