TASK 303: Musical Instruments Store
CLASS INSTRUMENT
    ATTRIBUTES
        PRIVATE brand: TEXT
        PRIVATE price: NUMBER
    CONSTRUCTOR(instrumentBrand: TEXT, instrumentPrice: NUMBER)
        SET brand TO instrumentBrand
        IF instrumentPrice GREATER THAN 0 THEN
            SET price TO instrumentPrice
        ELSE
            SET price TO 0
            DISPLAY "INVALID PRICE"
        END IF
    END CONSTRUCTOR
    METHODS
        GET_BRAND()
            RETURN brand
        END METHOD
        GET_PRICE()
            RETURN price
        END METHOD
        DESCRIBE_PLAY_STYLE()
            RETURN "GENERAL INSTRUMENT PLAY STYLE"
        END METHOD
END CLASS
CLASS GUITAR INHERITS INSTRUMENT
    ATTRIBUTES
        PRIVATE number_of_strings: NUMBER

        END IF
    END CONSTRUCTOR
    METHODS

        END METHOD
END CLASS

CLASS MUSIC_STORE
    ATTRIBUTES
        PRIVATE instruments: LIST<INSTRUMENT>[]
    CONSTRUCTOR()
        SET instruments TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        ADD_INSTRUMENT(newInstrument: INSTRUMENT)
            ADD newInstrument TO instruments
            DISPLAY "INSTRUMENT ADDED TO STORE"
        END METHOD
        VIEW_STOCK()
            REPEAT FROM I = 0 TO SIZE OF instruments - 1
                DISPLAY instruments[I].GET_BRAND()
                DISPLAY instruments[I].GET_PRICE()
                DISPLAY instruments[I].DESCRIBE_PLAY_STYLE()
            END REPEAT
        END METHOD
        CALCULATE_TOTAL_STOCK_VALUE()
            SET total TO 0
            REPEAT FROM I = 0 TO SIZE OF instruments - 1
                SET total TO total + instruments[I].GET_PRICE()
            END REPEAT
            RETURN total
        END METHOD
END CLASS
START
    BEGIN
        CREATE OBJECT store1 OF MUSIC_STORE()
        CREATE OBJECT guitar1 OF GUITAR("Yamaha", 120, 6)
        CREATE OBJECT drum1 OF DRUM_KIT("Pearl", 300, 5)
        CREATE OBJECT keyboard1 OF KEYBOARD("Casio", 180, 61)
        CALL store1.ADD_INSTRUMENT(guitar1)
        CALL store1.ADD_INSTRUMENT(drum1)
        CALL store1.ADD_INSTRUMENT(keyboard1)
        CALL store1.VIEW_STOCK()
        DISPLAY store1.CALCULATE_TOTAL_STOCK_VALUE()
    END
STOP

Task 306: Utility Billing System   

START 
   BEGAIN 
    CLASS METER_READING
      ATTRIBUTES
        PRIVATE start_reading: NUMBER
        PRIVATE end_reading: NUMBER     
        CONSTRUCTOR(startValue: NUMBER, endValue: NUMBER)
        IF endValue GREATER THAN OR EQUAL TO startValue THEN
            SET start_reading TO startValue
            SET end_reading TO endValue
        ELSE
            SET start_reading TO 0
            SET end_reading TO 0
            DISPLAY "INVALID METER READING"
        END IF
       END CONSTRUCTOR  
       METHODS
        CALCULATE_CONSUMPTION()
            RETURN end_reading - start_reading
        END METHOD
     END CLASS

  CLASS BILL
    ATTRIBUTES
        PRIVATE billing_period: TEXT
        PRIVATE due_date: TEXT
        PRIVATE meter_reading: METER_READING

    CONSTRUCTOR(period: TEXT, dueDate: TEXT, reading: METER_READING)
        SET billing_period TO period
        SET due_date TO dueDate
        SET meter_reading TO reading
    END CONSTRUCTOR
    METHODS
        GET_BILLING_PERIOD()
            RETURN billing_period
        END METHOD
        GET_METER_READING()
            RETURN meter_reading
        END METHOD
        CALCULATE_AMOUNT_OWED()
            RETURN 0
        END METHOD
    END CLASS
    CLASS ELECTRICITY_BILL INHERITS BILL
    ATTRIBUTES
        PRIVATE day_rate: NUMBER
        PRIVATE night_rate: NUMBER
        PRIVATE day_units: NUMBER
        PRIVATE night_units: NUMBER

    CONSTRUCTOR(period: TEXT, dueDate: TEXT, reading: METER_READING, dayRate: NUMBER, nightRate: NUMBER, dayUnitsValue: NUMBER, nightUnitsValue: NUMBER)
        CALL SUPER(period, dueDate, reading)

        SET day_rate TO dayRate
        SET night_rate TO nightRate
        SET day_units TO dayUnitsValue
        SET night_units TO nightUnitsValue
    END CONSTRUCTOR

    METHODS
        CALCULATE_AMOUNT_OWED()
            RETURN (day_units * day_rate) + (night_units * night_rate)
        END METHOD
END CLASS

CLASS WATER_BILL INHERITS BILL
    ATTRIBUTES
        PRIVATE flat_rate: NUMBER
        PRIVATE standing_charge: NUMBER

    CONSTRUCTOR(period: TEXT, dueDate: TEXT, reading: METER_READING, rate: NUMBER, charge: NUMBER)
        CALL SUPER(period, dueDate, reading)

        SET flat_rate TO rate
        SET standing_charge TO charge
    END CONSTRUCTOR

    METHODS
        CALCULATE_AMOUNT_OWED()
            RETURN GET_METER_READING().CALCULATE_CONSUMPTION() * flat_rate + standing_charge
        END METHOD
END CLASS

   END

CLASS STANDARD_RIDE INHERITS RIDE
    ATTRIBUTES

    CONSTRUCTOR(pickup: TEXT, dropoff: TEXT, rideRoute: ROUTE, rideDistance: NUMBER, rate: NUMBER)

        IF rate GREATER THAN 0 THEN
            SET rate_per_km TO rate
        ELSE
            SET rate_per_km TO 0
        END IF
    END CONSTRUCTOR
    METHODS
        CALCULATE_FARE()
            RETURN GET_DISTANCE() * rate_per_km
        END METHOD
END ClAS
CLAS SHARED_RIDE INHERITS STANDARD_RIDE
    ATTRIBUTES

    CONSTRUCTOR(pickup: TEXT, dropoff: TEXT, rideRoute: ROUTE, rideDistance: NUMBER, rate: NUMBER, passengers: NUMBER)
