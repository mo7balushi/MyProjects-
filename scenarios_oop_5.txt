CLASS CLAIM
    ATTRIBUTES
        PRIVATE claim_number: NUMBER
        PRIVATE date_filed: TEXT
        PRIVATE status: TEXT
        PRIVATE documentation_folder: DOCUMENTATION_FOLDER

    CONSTRUCTOR(claimNumber: NUMBER, filedDate: TEXT)
        SET claim_number TO claimNumber
        SET date_filed TO filedDate
        SET status TO "FILED"
        CREATE OBJECT folder OF DOCUMENTATION_FOLDER()
        SET documentation_folder TO folder
    END CONSTRUCTOR
    METHODS
        GET_STATUS()
            RETURN status
        END METHOD
        GET_DOCUMENTATION_FOLDER()
            RETURN documentation_folder
        END METHOD
        MOVE_TO_REVIEW()
            IF status EQUALS "FILED" THEN
                SET status TO "UNDER_REVIEW"
            ELSE
                DISPLAY "INVALID STATUS CHANGE"
            END IF
        END METHOD
        APPROVE()
            IF status EQUALS "UNDER_REVIEW" THEN
                SET status TO "APPROVED"
            ELSE
                DISPLAY "INVALID STATUS CHANGE"
            END IF
        END METHOD
        REJECT()
            IF status EQUALS "UNDER_REVIEW" THEN
                SET status TO "REJECTED"
            ELSE
                DISPLAY "INVALID STATUS CHANGE"
            END IF
        END METHOD
        CALCULATE_PAYOUT()
            RETURN 0
        END METHOD
END CLASS
CLASS VEHICLE_CLAIM INHERITS CLAIM
    ATTRIBUTES
        PRIVATE repair_estimate: NUMBER
        PRIVATE fixed_excess: NUMBER
    CONSTRUCTOR(claimNumber: NUMBER, filedDate: TEXT, repairEstimate: NUMBER, excess: NUMBER)
        CALL SUPER(claimNumber, filedDate)
        IF repairEstimate GREATER THAN 0 THEN
            SET repair_estimate TO repairEstimate
        ELSE
            SET repair_estimate TO 0
        END IF
        IF excess GREATER THAN 0 THEN
            SET fixed_excess TO excess
        ELSE
            SET fixed_excess TO 0
        END IF
    END CONSTRUCTOR
    METHODS
        CALCULATE_PAYOUT()
            SET payout TO repair_estimate - fixed_excess
            IF payout LESS THAN 0 THEN
                RETURN 0
            ELSE
                RETURN payout
            END IF
        END METHOD
END CLASS

        PRIVATE location: TEXT
        PRIVATE servers: LIST<SERVER>[]
    CONSTRUCTOR(rackId: NUMBER, rackLocation: TEXT)
        SET rack_id TO rackId
        SET location TO rackLocation
        SET servers TO EMPTY LIST
    END CONSTRUCTOR
    METHODS
        ADD_SERVER(newServer: SERVER)
            ADD newServer TO servers
            DISPLAY "SERVER ADDED TO RACK"
        END METHOD
        CALCULATE_AVERAGE_LOAD()
            IF SIZE OF servers EQUALS 0 THEN
                RETURN 0
            END IF
            SET total TO 0
            REPEAT FROM I = 0 TO SIZE OF servers - 1
                SET total TO total + servers[I].CALCULATE_LOAD_PERCENTAGE()
            END REPEAT
            RETURN total / SIZE OF servers
        END METHOD
END CLASS

        CREATE OBJECT hardware3 OF HARDWARE_SPECIFICATION("Intel Xeon", "256GB", "8TB SSD")

        CREATE OBJECT dbServer1 OF DATABASE_SERVER("DB001", 2000, hardware2, 800, 1000)
        CREATE OBJECT cacheServer1 OF CACHE_SERVER("CACHE001", 1500, hardware3, 800, 1000, 0.75)
        CREATE OBJECT rack1 OF RACK(1, "Rack A")
        CALL rack1.ADD_SERVER(webServer1)
        CALL rack1.ADD_SERVER(dbServer1)
        CALL rack1.ADD_SERVER(cacheServer1)

        DISPLAY webServer1.CALCULATE_LOAD_PERCENTAGE()
        DISPLAY dbServer1.CALCULATE_LOAD_PERCENTAGE()
        DISPLAY cacheServer1.CALCULATE_LOAD_PERCENTAGE()
        DISPLAY rack1.CALCULATE_AVERAGE_LOAD()
    END
