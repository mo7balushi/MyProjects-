TASK 293: Shape Drawing App

CLASS SHAPE
    ATTRIBUTES
        PRIVATE color: TEXT
        PRIVATE x: NUMBER
        PRIVATE y: NUMBER

    CONSTRUCTOR(shapeColor: TEXT, xValue: NUMBER, yValue: NUMBER)
        SET this.color TO shapeColor
        SET this.x TO xValue
        SET this.y TO yValue
    END CONSTRUCTOR

    METHODS
        GET_COLOR()
            RETURN this.color
        END METHOD

        GET_POSITION()
            DISPLAY "X: " + this.x
            DISPLAY "Y: " + this.y
        END METHOD

        GET_AREA()
            DISPLAY "GET_AREA must be overridden by child class"
            RETURN 0
        END METHOD

        GET_PERIMETER()
            DISPLAY "GET_PERIMETER must be overridden by child class"
            RETURN 0
        END METHOD
END CLASS


CLASS CIRCLE EXTENDS SHAPE
    ATTRIBUTES
        PRIVATE radius: NUMBER

    CONSTRUCTOR(shapeColor: TEXT, xValue: NUMBER, yValue: NUMBER, circleRadius: NUMBER)
        CALL SUPER(shapeColor, xValue, yValue)

        IF circleRadius GREATER THAN 0 THEN
            SET this.radius TO circleRadius
        ELSE
            SET this.radius TO 0
            DISPLAY "INVALID RADIUS"
        END IF
    END CONSTRUCTOR

    METHODS
        GET_AREA()
            RETURN 3.14 * this.radius * this.radius
        END METHOD

        GET_PERIMETER()
            RETURN 2 * 3.14 * this.radius
        END METHOD
END CLASS


CLASS RECTANGLE EXTENDS SHAPE
    ATTRIBUTES
        PRIVATE width: NUMBER
        PRIVATE height: NUMBER

    CONSTRUCTOR(shapeColor: TEXT, xValue: NUMBER, yValue: NUMBER, rectWidth: NUMBER, rectHeight: NUMBER)
        CALL SUPER(shapeColor, xValue, yValue)

        IF rectWidth GREATER THAN 0 THEN
            SET this.width TO rectWidth
        ELSE
            SET this.width TO 0
            DISPLAY "INVALID WIDTH"
        END IF

        IF rectHeight GREATER THAN 0 THEN
            SET this.height TO rectHeight
        ELSE
            SET this.height TO 0
            DISPLAY "INVALID HEIGHT"
        END IF
    END CONSTRUCTOR

    METHODS
        GET_AREA()
            RETURN this.width * this.height
        END METHOD

        GET_PERIMETER()
            RETURN 2 * (this.width + this.height)
        END METHOD
END CLASS


CLASS TRIANGLE EXTENDS SHAPE
    ATTRIBUTES
        PRIVATE side_one: NUMBER
        PRIVATE side_two: NUMBER
        PRIVATE side_three: NUMBER

    CONSTRUCTOR(shapeColor: TEXT, xValue: NUMBER, yValue: NUMBER, s1: NUMBER, s2: NUMBER, s3: NUMBER)
        CALL SUPER(shapeColor, xValue, yValue)

        IF s1 GREATER THAN 0 THEN
            SET this.side_one TO s1
        ELSE
            SET this.side_one TO 0
        END IF

        IF s2 GREATER THAN 0 THEN
            SET this.side_two TO s2
        ELSE
            SET this.side_two TO 0
        END IF

        IF s3 GREATER THAN 0 THEN
            SET this.side_three TO s3
        ELSE
            SET this.side_three TO 0
        END IF
    END CONSTRUCTOR

    METHODS
        GET_PERIMETER()
            RETURN this.side_one + this.side_two + this.side_three
        END METHOD

        GET_AREA()
            SET semiPerimeter TO GET_PERIMETER() / 2

            RETURN SQUARE_ROOT(semiPerimeter * (semiPerimeter - this.side_one) * (semiPerimeter - this.side_two) * (semiPerimeter - this.side_three))
        END METHOD
END CLASS


CLASS CANVAS
    ATTRIBUTES
        PRIVATE shapes: LIST<SHAPE>[]

    CONSTRUCTOR()
        SET this.shapes TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_SHAPE(newShape: SHAPE)
            ADD newShape TO this.shapes
            DISPLAY "SHAPE ADDED TO CANVAS"
        END METHOD

        CALCULATE_TOTAL_AREA()
            SET total TO 0

            REPEAT FROM I = 0 TO SIZE OF this.shapes - 1
                SET total TO total + this.shapes[I].GET_AREA()
            END REPEAT

            RETURN total
        END METHOD
END CLASS


START
    BEGIN
        CREATE OBJECT canvas1 OF CANVAS()

        CREATE OBJECT circle1 OF CIRCLE("Red", 10, 20, 5)
        CREATE OBJECT rectangle1 OF RECTANGLE("Blue", 30, 40, 10, 6)
        CREATE OBJECT triangle1 OF TRIANGLE("Green", 50, 60, 3, 4, 5)

        CALL canvas1.ADD_SHAPE(circle1)
        CALL canvas1.ADD_SHAPE(rectangle1)
        CALL canvas1.ADD_SHAPE(triangle1)

        DISPLAY canvas1.CALCULATE_TOTAL_AREA()
    END
STOP

TASK 296: Online Payment Methods

CLASS PAYMENT_METHOD
    ATTRIBUTES
        PRIVATE amount: NUMBER
        PRIVATE status: TEXT

    CONSTRUCTOR(paymentAmount: NUMBER)
        IF paymentAmount GREATER THAN 0 THEN
            SET this.amount TO paymentAmount
        ELSE
            SET this.amount TO 0
            DISPLAY "INVALID AMOUNT"
        END IF

        SET this.status TO "PENDING"
    END CONSTRUCTOR

    METHODS
        GET_AMOUNT()
            RETURN this.amount
        END METHOD

        PROCESS_PAYMENT()
            DISPLAY "PROCESS_PAYMENT must be overridden by child class"
            RETURN FALSE
        END METHOD
END CLASS


CLASS CREDIT_CARD_PAYMENT EXTENDS PAYMENT_METHOD
    ATTRIBUTES
        PRIVATE card_number: TEXT
        PRIVATE expiry_date: TEXT

    CONSTRUCTOR(amount: NUMBER, cardNumber: TEXT, expiryDate: TEXT)
        CALL SUPER(amount)
        SET this.card_number TO cardNumber
        SET this.expiry_date TO expiryDate
    END CONSTRUCTOR

    METHODS
        VALIDATE_CARD()
            IF LENGTH OF this.card_number EQUALS 16 AND this.expiry_date NOT EQUAL TO "" THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        PROCESS_PAYMENT()
            IF VALIDATE_CARD() EQUALS TRUE THEN
                DISPLAY "CREDIT CARD PAYMENT APPROVED"
                RETURN TRUE
            ELSE
                DISPLAY "CREDIT CARD PAYMENT REJECTED"
                RETURN FALSE
            END IF
        END METHOD
END CLASS

    ATTRIBUTES

    CONSTRUCTOR(amount: NUMBER, account: TEXT)
        CALL SUPER(amount)
        SET this.bank_account TO account
    END CONSTRUCTOR

    METHODS

            IF this.bank_account NOT EQUAL TO "" THEN
                DISPLAY "BANK TRANSFER PAYMENT APPROVED"
                RETURN TRUE
            ELSE
                DISPLAY "BANK TRANSFER PAYMENT REJECTED"
                RETURN FALSE
            END IF
        END METHOD
END CLASS


CLASS WALLET_PAYMENT EXTENDS PAYMENT_METHOD
    ATTRIBUTES
        PRIVATE wallet_balance: NUMBER

    CONSTRUCTOR(amount: NUMBER, balance: NUMBER)
        CALL SUPER(amount)
        SET this.wallet_balance TO balance
    END CONSTRUCTOR

    METHODS
        PROCESS_PAYMENT()
            IF this.wallet_balance GREATER THAN OR EQUAL TO GET_AMOUNT() THEN
                SET this.wallet_balance TO this.wallet_balance - GET_AMOUNT()
                DISPLAY "WALLET PAYMENT APPROVED"
                RETURN TRUE
            ELSE
                DISPLAY "INSUFFICIENT WALLET BALANCE"
                RETURN FALSE
            END IF
        END METHOD
END CLASS

CLASS CREDIT_CARD EXTENDS CARD

    CONSTRUCTOR(cardNumber: TEXT, holderName: TEXT, cardChip: CHIP, limitValue: NUMBER)
        CALL SUPER(cardNumber, holderName, cardChip)

        SET this.credit_limit TO limitValue
        SET this.unpaid_purchases TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        GET_OUTSTANDING_BALANCE()
            SET total TO 0

            REPEAT FROM I = 0 TO SIZE OF this.unpaid_purchases - 1
                SET total TO total + this.unpaid_purchases[I]
            END REPEAT

            RETURN total
        END METHOD

        AUTHORISE_PURCHASE(amount: NUMBER)
            IF IS_BLOCKED() EQUALS TRUE THEN
                RETURN FALSE
            END IF

            IF GET_OUTSTANDING_BALANCE() + amount LESS THAN OR EQUAL TO this.credit_limit THEN
                ADD amount TO this.unpaid_purchases
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        MAKE_PAYMENT(amount: NUMBER)
            IF amount GREATER THAN 0 THEN
                SET remaining TO amount

                REPEAT FROM I = 0 TO SIZE OF this.unpaid_purchases - 1
                    IF remaining GREATER THAN 0 THEN
                        SET this.unpaid_purchases[I] TO this.unpaid_purchases[I] - remaining

                        IF this.unpaid_purchases[I] LESS THAN 0 THEN
                            SET remaining TO 0 - this.unpaid_purchases[I]
                            SET this.unpaid_purchases[I] TO 0
                        ELSE
                            SET remaining TO 0
                        END IF
                    END IF
                END REPEAT
            END IF
        END METHOD
END CLASS

        END METHOD

        CHECK_OVERLAP(otherShift: SHIFT)

                RETURN TRUE
            ELSE
