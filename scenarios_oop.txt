Scenario1 :

CLASS Hospital
    ATTRIBUTES
        id : NUMBER
        name : STRING
        address : STRING
        patients : LIST OF Patient
    METHODS
        ADMIT_PATIENT(patient : Patient)
            ADD patient TO patients
            SET patient.isAdmitted TO true
            DISPLAY "Patient admitted successfully"
        END METHOD
        DISCHARGE_PATIENT(patientId : NUMBER, dischargeDate : STRING)
            FOR EACH patient IN patients
                IF patient.id = patientId THEN
                    SET patient.isAdmitted TO false
                    SET patient.dischargeDate TO dischargeDate
                    DISPLAY "Patient discharged successfully"
                END IF
            END FOR
        END METHOD
        COUNT_ADMITTED_PATIENTS()
            SET count TO 0
            FOR EACH patient IN patients
                IF patient.isAdmitted = true THEN
                    SET count TO count + 1
                END IF
            END FOR
            DISPLAY count
        END METHOD
END CLASS




Scenario2 

CLAS Patient
    ATTRIBUTES
        id : NUMBER
        fullName : STRING
        dateOfBirth : STRING
        admissionDate : STRING
        dischargeDate : STRING
        isAdmitted : BOOLEAN
        room : Room
        procedures : LIST OF Procedure
    METHODS
        ADD_PROCEDURE(procedure : Procedure)
            ADD procedure TO procedures
            DISPLAY "Procedure added successfully"
        END METHOD
        MOVE_TO_ROOM(newRoom : Room)
            SET room TO newRoom
            DISPLAY "Patient moved to new room"
        END METHOD
        TOTAL_COST_OF_CARE()
            SET total TO 0
            FOR EACH procedure IN procedures
                SET total TO total + procedure.cost
            END FOR
            DISPLAY "Total cost is " + total
        END METHOD
        DISPLAY_DETAILS()
            DISPLAY fullName
            DISPLAY admissionDate
            DISPLAY dischargeDate
            DISPLAY room.roomNumber
        END METHOD
END CLASS
